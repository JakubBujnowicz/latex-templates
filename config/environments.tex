% Bolded title -----------------------------------------------------------------
\theoremstyle{definition}

% Definition
\newtheorem{definition}{\definitionname}[section]
\newtheorem{bdefinition}{\definitionname}[section]

% Example
\newtheorem{example}{\examplename}[section]
\newtheorem{bexample}{\examplename}[section]


% Bolded title, italic text ----------------------------------------------------
\theoremstyle{plain}

% Theorem
\newtheorem{theorem}{\theoremname}[section]
\newtheorem{btheorem}{\theoremname}[section]

% Corollary
\newtheorem{corollary}{\corollaryname}[section]
\newtheorem{bcorollary}{\corollaryname}[section]

% Lemma
\newtheorem{lemma}{\lemmaname}[section]
\newtheorem{blemma}{\lemmaname}[section]


% Italic title -----------------------------------------------------------------
\theoremstyle{remark}

% Remark
\newtheorem{remark}{\remarkname}[section]
\newtheorem{bremark}{\remarkname}[section]
